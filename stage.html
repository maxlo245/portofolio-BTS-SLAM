  • script_traitement_fichiers.py : Copie automatisée de fichiers d'un dossier source vers un dossier de destination, utile pour la gestion documentaire.
    Télécharger le code
    script_traitement_fichiers.py 
    import os
import shutil

def copier_fichiers(source, destination):
    if not os.path.exists(destination):
        os.makedirs(destination)
    for fichier in os.listdir(source):
        chemin_fichier = os.path.join(source, fichier)
        if os.path.isfile(chemin_fichier):
            shutil.copy2(chemin_fichier, destination)

if __name__ == "__main__":
    dossier_source = r"C:\\Users\\maxim\\Documents\\source"
    dossier_destination = r"C:\\Users\\maxim\\Documents\\destination"
    copier_fichiers(dossier_source, dossier_destination)
    print("Copie terminée.")
